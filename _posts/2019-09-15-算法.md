1.冒泡排序，相邻数据两两比较，把较大数升到最顶端

~~~
function pop(arr) {
  for(let i = arr.length - 1;i > 1;i--) {
    for (let j = 0;j < i;j++) {
      if (arr[j] > arr[j+1]) {
        let tem = arr[j];
        arr[j] = arr[j + 1];
        arr[j+1] = tem;
      }
    }
  }
  return arr;
}
~~~

2.选择算法，从底端开始比较数据大小，确定最大值，记录数组index位置，然后与顶端值交换

~~~
function choice(arr) {
  for(let i = arr.length - 1;i >= 1;i--) {
    let max = 0;
    for (let j = 1;j <= i;j++) {
      if (arr[j] > arr[max]) {
        max = j;
      }
    }
    let tem = arr[max];
    arr[max] = arr[i];
    arr[i] = tem;
  }
  return arr;
}
~~~

4.希尔排序，特殊的插入排序，定义间隔序列，用间隔进行排序，然后再进行插入排序

~~~
function xier(arr, gap) {
  for (var i = 0; i< gap.length;i++) {
    for(var j = gap[i]; j<arr.length;j++) {
      var tem = arr[j];
      for(var k = j-gap[i];k > -1;k-=gap[i]) {
        if (arr[k] > tem) {
          arr[k+gap[i]] = arr[k];
          if (k - gap[i] < 0) {
            arr[k] = tem;
          }
        } else {
          arr[k + gap[i]] = tem;
          break;
        }
      }
    }
  }
  return arr;
}

5.归并排序，将大数组分为小数组排序，空间换时间，两种方式，自下而上的迭代和自上而下的递归,递归容易造成栈溢出

~~~
// 递归
function sort(arr) {
  if (arr.length < 2) {
    return arr;
  }
  var left = arr.slice(0, Math.floor(arr.length/2));
  var right = arr.slice(Math.floor(arr.length/2));
  return merge(sort(left),sort(right));
}
function merge(left,right) {
  let result = [];
  while(left.length && right.length) {
    if(left[0] < right[0]) {
      result.push(left.shift());
    } else {
      result.push(right.shift());
    }
  }
  while(left.length) {
    result.push(left.shift());
  }
  while(right.length) {
    result.push(right.shift());
  }
  return result;
}
